    export namespace Utilities.Popup {
        export function showVirusScanProgessPopup(content: string, options: any) {
            let $okButton = createButton(ICM.MessageLocal.OkLabel, function (e: any): void {
                closePopup($popupDiv);
                options.okButtonClicked();
            }, true);

            $okButton.attr('data-bind', 'enabled: isChecked');

            let $popupDiv = createPopupTemplate('popupVirusScan', content, [$okButton], createCloseButton(null));
            $popupDiv.removeClass().addClass('popup_block confBox commitmentslabels modal fade in');

            showPopup($popupDiv);
            kendo.bind($popupDiv, options);
        }
    }
